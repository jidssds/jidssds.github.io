<script>
// Default float buddy
CursorCompanion.add({ followSpeed: 0.2, idleEffects: { type: 'float', intensity: 6 } });

// Bounce buddy
CursorCompanion.add({
  element: (() => { const d=document.createElement('div'); d.style.width='40px'; d.style.height='40px'; d.style.borderRadius='50%'; d.style.background='red'; document.body.appendChild(d); return d; })(),
  followSpeed: 0.15,
  hoverEffects: { type:'bounce', intensity:1.2 }
});

// Squash buddy
CursorCompanion.add({

  followSpeed: 0.12,
  clickEffects: { type:'squash', intensity:0.8 }
});
